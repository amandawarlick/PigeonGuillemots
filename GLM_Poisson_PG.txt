{
    alpha ~ dnorm(0.00000E+00, 1.00000E-04)
    beta1 ~ dnorm(0.00000E+00, 1.00000E-04)
    for (i in 1:n) {
        C[i] ~ dpois(lambda[i])
        log(lambda[i]) <- log.lambda[i]
        log.lambda[i] <- alpha + beta1 * year[i]
        resi[i] <- (C[i] - lambda[i])/(sqrt(lambda[i]) + e)
    }
}
